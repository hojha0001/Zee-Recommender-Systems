from scipy.stats import pearsonr

# Create synthetic ratings data (since we don't have real ratings)
np.random.seed(42)
n_users = 1000  # More users for better recommendations
n_movies = len(movies_df)

# Create more realistic ratings data
ratings_data = []
for user_id in range(1, n_users + 1):
    # Each user rates between 20 and 100 movies
    n_ratings = np.random.randint(20, 100)
    movie_indices = np.random.choice(n_movies, n_ratings, replace=False)
    
    for movie_idx in movie_indices:
        # Ratings between 1 and 5, with half-star precision
        rating = np.random.choice(np.arange(1, 5.5, 0.5))
        ratings_data.append({
            'UserID': user_id,
            'MovieID': movies_df.iloc[movie_idx]['MovieID'],
            'Rating': rating
        })

ratings_df = pd.DataFrame(ratings_data)

# Create the ratings matrix (items x users)
ratings_matrix = ratings_df.pivot(index='MovieID', columns='UserID', values='Rating')

def compute_similarity(ratings_matrix):
    """Compute the similarity matrix using Pearson correlation"""
    similarity_matrix = pd.DataFrame(index=ratings_matrix.index, columns=ratings_matrix.index)
    
    for i in ratings_matrix.index:
        for j in ratings_matrix.index:
            item1 = ratings_matrix.loc[i]
            item2 = ratings_matrix.loc[j]
            
            # Find common ratings
            common_users = item1.notna() & item2.notna()
            
            if common_users.sum() > 5:  # Only compute if there are at least 5 common ratings
                correlation, _ = pearsonr(item1[common_users], item2[common_users])
                similarity_matrix.loc[i,j] = correlation
            else:
                similarity_matrix.loc[i,j] = 0
                
    return similarity_matrix

def get_recommendations(movie_id, similarity_matrix, n_recommendations=5):
    """Get movie recommendations based on similarity"""
    if movie_id not in similarity_matrix.index:
        return "Movie ID not found"
    
    # Get similar movies
    similar_scores = similarity_matrix.loc[movie_id].sort_values(ascending=False)
    # Remove the movie itself and get top n
    similar_scores = similar_scores[similar_scores.index != movie_id][:n_recommendations]
    
    # Create recommendations dataframe
    recommendations = pd.DataFrame({
        'MovieID': similar_scores.index,
        'Similarity': similar_scores.values
    })
    
    # Add movie information
    recommendations = recommendations.merge(movies_df[['MovieID', 'Title', 'Genres']], on='MovieID')
    
    return recommendations

5. Build a Recommender System based on Matrix Factorization.

@ Create a Recommender System using the Matrix Factorization method

Matrix factorization is a popular method for creating recommender systems, particularly useful for collaborative filtering. It decomposes the user-item interaction matrix into two lower-dimensional matrices one representing the users and the other representing the items which can then be used to predict missing values (ratings).

A commonly used matrix factorization technique is Singular Value Decomposition (SVD), which is implemented in libraries such as scikit-surprise and scikit-learn.
